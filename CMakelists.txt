function(maybe_link_target tgt exe_target)
    if(TARGET ${tgt})
        get_target_property(_tgt_type ${tgt} TYPE)
        if(NOT _tgt_type STREQUAL "UTILITY")
            message(STATUS "Linking ${exe_target} with target: ${tgt} (type=${_tgt_type})")
            target_link_libraries(${exe_target} PRIVATE ${tgt})
        endif()
    endif()
endfunction()

# -------------------------------------------------------------------------
# Qt prebuilt fallback: try find_package first, otherwise create IMPORTED targets from .a files
# -------------------------------------------------------------------------
set(THIRD_PARTY_QT_DIR "${CMAKE_SOURCE_DIR}/Thirdparty/prebuilt/qt/windows/6.10.2/mingw_64")
if(EXISTS "${THIRD_PARTY_QT_DIR}")
    message(STATUS "Adding Qt prebuilt path to CMAKE_PREFIX_PATH: ${THIRD_PARTY_QT_DIR}")
    list(APPEND CMAKE_PREFIX_PATH "${THIRD_PARTY_QT_DIR}")
endif()

# Try config-mode find first
find_package(Qt6 COMPONENTS Core Gui Widgets QUIET)
if(TARGET Qt6::Core)
    message(STATUS "Qt6 found via find_package; using Qt6::Core/Gui/Widgets")
else()
    # fallback: create imported targets from static libs under prebuilt path
    set(QT_PREBUILT_DIR "${THIRD_PARTY_QT_DIR}")
    if(EXISTS "${QT_PREBUILT_DIR}")
        message(STATUS "Attempting to create imported Qt targets from: ${QT_PREBUILT_DIR}")

        set(_qt_lib_dirs
            "${QT_PREBUILT_DIR}/lib"
            "${QT_PREBUILT_DIR}/lib64"
            "${QT_PREBUILT_DIR}/lib/x64"
            "${QT_PREBUILT_DIR}/lib/x86"
            "${QT_PREBUILT_DIR}"
        )

        function(_find_qt_lib outvar)
            set(_names ${ARGN})
            set(_found "")
            foreach(n ${_names})
                find_library(_tmp_lib NAMES ${n} PATHS ${_qt_lib_dirs} NO_DEFAULT_PATH)
                if(_tmp_lib)
                    set(_found "${_tmp_lib}")
                    break()
                endif()
            endforeach()
            set(${outvar} "${_found}" PARENT_SCOPE)
        endfunction()

        _find_qt_lib(_qt_core_lib Qt6Core Qt6Core4 libQt6Core Qt6Core_a)
        _find_qt_lib(_qt_gui_lib  Qt6Gui  Qt6Gui4  libQt6Gui  Qt6Gui_a)
        _find_qt_lib(_qt_widgets_lib Qt6Widgets Qt6Widgets4 libQt6Widgets Qt6Widgets_a)

        # try to detect include dir: prefer include/ or include/Qt6 etc.
        set(_qt_include_dir "")
        if(EXISTS "${QT_PREBUILT_DIR}/include")
            set(_qt_include_dir "${QT_PREBUILT_DIR}/include")
        else()
            file(GLOB_RECURSE _qapp_files LIST_DIRECTORIES false "${QT_PREBUILT_DIR}/**/QApplication" "${QT_PREBUILT_DIR}/**/QApplication.h")
            if(_qapp_files)
                list(GET _qapp_files 0 _first_qapp)
                get_filename_component(_qt_include_dir "${_first_qapp}" DIRECTORY)
            endif()
        endif()

        if(_qt_core_lib AND NOT TARGET Qt6::Core)
            add_library(Qt6::Core STATIC IMPORTED)
            set_target_properties(Qt6::Core PROPERTIES
                IMPORTED_LOCATION "${_qt_core_lib}"
                IMPORTED_LOCATION_RELEASE "${_qt_core_lib}"
                IMPORTED_LOCATION_DEBUG "${_qt_core_lib}"
            )
            if(_qt_include_dir)
                set_target_properties(Qt6::Core PROPERTIES INTERFACE_INCLUDE_DIRECTORIES "${_qt_include_dir}")
                message(STATUS "Qt6::Core include dir set to: ${_qt_include_dir}")
            endif()
            message(STATUS "Created IMPORTED Qt6::Core -> ${_qt_core_lib}")
        endif()

        if(_qt_gui_lib AND NOT TARGET Qt6::Gui)
            add_library(Qt6::Gui STATIC IMPORTED)
            set_target_properties(Qt6::Gui PROPERTIES
                IMPORTED_LOCATION "${_qt_gui_lib}"
                IMPORTED_LOCATION_RELEASE "${_qt_gui_lib}"
                IMPORTED_LOCATION_DEBUG "${_qt_gui_lib}"
            )
            if(_qt_include_dir)
                set_target_properties(Qt6::Gui PROPERTIES INTERFACE_INCLUDE_DIRECTORIES "${_qt_include_dir}")
            endif()
            if(TARGET Qt6::Core)
                set_target_properties(Qt6::Gui PROPERTIES INTERFACE_LINK_LIBRARIES "Qt6::Core")
            endif()
            message(STATUS "Created IMPORTED Qt6::Gui -> ${_qt_gui_lib}")
        endif()

        if(_qt_widgets_lib AND NOT TARGET Qt6::Widgets)
            add_library(Qt6::Widgets STATIC IMPORTED)
            set_target_properties(Qt6::Widgets PROPERTIES
                IMPORTED_LOCATION "${_qt_widgets_lib}"
                IMPORTED_LOCATION_RELEASE "${_qt_widgets_lib}"
                IMPORTED_LOCATION_DEBUG "${_qt_widgets_lib}"
            )
            if(_qt_include_dir)
                set_target_properties(Qt6::Widgets PROPERTIES INTERFACE_INCLUDE_DIRECTORIES "${_qt_include_dir}")
            endif()

            set(_iface_libs "")
            if(TARGET Qt6::Gui)
                list(APPEND _iface_libs "Qt6::Gui")
            elseif(_qt_gui_lib)
                list(APPEND _iface_libs "${_qt_gui_lib}")
            endif()
            if(TARGET Qt6::Core)
                list(APPEND _iface_libs "Qt6::Core")
            elseif(_qt_core_lib)
                list(APPEND _iface_libs "${_qt_core_lib}")
            endif()
            if(_iface_libs)
                set_target_properties(Qt6::Widgets PROPERTIES INTERFACE_LINK_LIBRARIES "${_iface_libs}")
            endif()
            message(STATUS "Created IMPORTED Qt6::Widgets -> ${_qt_widgets_lib}")
        endif()

        if(NOT _qt_include_dir)
            message(WARNING "Qt include dir not auto-detected under ${QT_PREBUILT_DIR}. You may need to set INTERFACE_INCLUDE_DIRECTORIES manually.")
            file(GLOB _qt_inc_cands "${QT_PREBUILT_DIR}/**/QtCore" "${QT_PREBUILT_DIR}/**/QtWidgets" "${QT_PREBUILT_DIR}/**/Qt6")
            if(_qt_inc_cands)
                message(STATUS "Found candidate include dirs (first 5):")
                list(LENGTH _qt_inc_cands _c)
                math(EXPR _max_show "(_c < 5) ? _c : 5")
                foreach(i RANGE 0 ${_max_show})
                    list(GET _qt_inc_cands ${i} _f)
                    message(STATUS "  ${_f}")
                endforeach()
            endif()
        endif()
    else()
        message(STATUS "Qt prebuilt dir not found at ${QT_PREBUILT_DIR}")
    endif()
endif()
